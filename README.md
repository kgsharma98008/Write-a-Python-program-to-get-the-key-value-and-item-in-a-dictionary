# Write-a-Python-program-to-get-the-key-value-and-item-in-a-dictionary

my_dict = {'apple': 5, 'banana': 3, 'orange': 2}
# get keys
for key in my_dict.keys():
 print('Key:', key)
# get values
for value in my_dict.values():
 print('Value:', value)
# get items
for item in my_dict.items():
 print('Item:', item)
